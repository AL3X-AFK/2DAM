CREATE OR REPLACE TYPE triangulo AS OBJECT(
	base NUMBER,
	altura NUMBER,
	MEMBER FUNCTION area RETURN NUMBER
);
/

CREATE OR REPLACE TYPE BODY triangulo AS
	MEMBER FUNCTION area RETURN NUMBER IS
	BEGIN
		RETURN (base*altura)/2;
	END;
END;
/ 

CREATE TABLE triangulos(
Id NUMBER,
tri triangulo);

INSERT INTO triangulos VALUES (1,triangulo(5,5));
INSERT INTO triangulos VALUES (2,triangulo(10,10));

SELECT * FROM triangulos;

DECLARE
	t triangulo;
	a NUMBER;
BEGIN
	FOR rec IN (SELECT Id, tri FROM triangulos) LOOP
	t:= rec.tri;
	a:= t.area;
	DBMS_OUTPUT.PUT_LINE('El triangulo con id: '||rec.Id);
	DBMS_OUTPUT.PUT_LINE('Con base: '||t.base);
	DBMS_OUTPUT.PUT_LINE('Y altura: '||t.altura);
	DBMS_OUTPUT.PUT_LINE('Tiene un area: '||a);
	DBMS_OUTPUT.PUT_LINE('------------------------------');
	END LOOP;
END;
/
